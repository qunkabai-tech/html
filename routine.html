    <header class="p-6 bg-indigo-600 text-white rounded-b-[2.5rem] shadow-lg sticky top-0 z-10">

        <div class="flex justify-between items-center mb-6">

            <div>

                <h1 class="text-xl font-bold">My 30 Days</h1>

                <p id="selected-date-display" class="text-indigo-100 text-sm">Loading...</p>

            </div>

            <div class="relative w-14 h-14">

                <svg class="w-full h-full">

                    <circle cx="28" cy="28" r="24" stroke="currentColor" stroke-width="4" fill="transparent" class="text-indigo-400" />

                    <circle id="progress-circle" cx="28" cy="28" r="24" stroke="white" stroke-width="4" fill="transparent" stroke-dasharray="150.8" stroke-dashoffset="150.8" class="progress-ring" />

                </svg>

                <div id="progress-text" class="absolute inset-0 flex items-center justify-center text-[10px] font-bold">0%</div>

            </div>

        </div>

        <div id="calendar-strip" class="flex gap-3 overflow-x-auto pb-2 scroll-hide"></div>

    </header>

    <main class="flex-1 p-5 space-y-4" id="task-container"></main>

    <footer class="p-4 bg-white border-t sticky bottom-0 shadow-[0_-4px_20px_-5px_rgba(0,0,0,0.1)]">

        <div class="flex gap-2">

            <input type="text" id="task-input" placeholder="Add a plan..." class="flex-1 p-3 bg-gray-50 border border-gray-200 rounded-2xl focus:outline-none focus:ring-2 focus:ring-indigo-500">

            <button onclick="addTask()" class="bg-indigo-600 text-white px-6 py-3 rounded-2xl font-bold active:scale-95 transition-transform">

                <i class="fas fa-plus"></i>

            </button>

        </div>

    </footer>

    <script>

        let allData = JSON.parse(localStorage.getItem('monthly_routines_v2')) || {};

        const START_DATE = new Date(2024, 3, 17); // April 17

        const END_DATE = new Date(2024, 4, 17);   // May 17

        let selectedDate = new Date().toISOString().split('T')[0];

        const taskContainer = document.getElementById('task-container');

        const taskInput = document.getElementById('task-input');

        const progressCircle = document.getElementById('progress-circle');

        const progressText = document.getElementById('progress-text');

        const calendarStrip = document.getElementById('calendar-strip');

        const selectedDateDisplay = document.getElementById('selected-date-display');

        function initCalendar() {

            calendarStrip.innerHTML = '';

            let current = new Date(START_DATE);

            while (current <= END_DATE) {

                const dateStr = current.toISOString().split('T')[0];

                const dayNum = current.getDate();

                const monthShort = new Intl.DateTimeFormat('en-US', { month: 'short' }).format(current);

                const weekday = new Intl.DateTimeFormat('en-US', { weekday: 'short' }).format(current);

                const dayEl = document.createElement('div');

                dayEl.className = `calendar-day flex-shrink-0 w-14 h-20 rounded-2xl flex flex-col items-center justify-center cursor-pointer transition-all relative ${dateStr === selectedDate ? 'active' : 'bg-indigo-50 text-indigo-900'}`;

                if (allData[dateStr]?.length > 0) dayEl.classList.add('has-tasks');

                dayEl.onclick = () => selectDate(dateStr);

                dayEl.innerHTML = `<span class="text-[9px] uppercase font-bold opacity-60">${weekday}</span><span class="text-lg font-bold">${dayNum}</span><span class="text-[9px] font-medium">${monthShort}</span>`;

                calendarStrip.appendChild(dayEl);

                if (dateStr === selectedDate) setTimeout(() => dayEl.scrollIntoView({ behavior: 'smooth', inline: 'center' }), 100);

                current.setDate(current.getDate() + 1);

            }

        }

        function selectDate(dateStr) {

            selectedDate = dateStr;

            selectedDateDisplay.innerText = new Date(dateStr).toLocaleDateString('en-US', { day: 'numeric', month: 'long', year: 'numeric' });

            initCalendar(); renderTasks(); updateProgress();

        }

        function save() { localStorage.setItem('monthly_routines_v2', JSON.stringify(allData)); renderTasks(); updateProgress(); }

        function addTask() {

            const text = taskInput.value.trim();

            if (!text) return;

            if (!allData[selectedDate]) allData[selectedDate] = [];

            allData[selectedDate].push({ id: Date.now(), text, completed: false });

            taskInput.value = ""; save(); initCalendar();

        }

        function toggleTask(id) {

            allData[selectedDate] = allData[selectedDate].map(t => t.id === id ? { ...t, completed: !t.completed } : t);

            save();

        }

        function deleteTask(id) {

            allData[selectedDate] = allData[selectedDate].filter(t => t.id !== id);

            save(); initCalendar();

        }

        function updateProgress() {

            const dayTasks = allData[selectedDate] || [];

            if (dayTasks.length === 0) { progressText.innerText = "0%"; progressCircle.style.strokeDashoffset = 150.8; return; }

            const percent = Math.round((dayTasks.filter(t => t.completed).length / dayTasks.length) * 100);

            progressText.innerText = `${percent}%`;

            progressCircle.style.strokeDashoffset = 150.8 - (percent / 100) * 150.8;

        }

        function renderTasks() {

            taskContainer.innerHTML = '';

            const dayTasks = allData[selectedDate] || [];

            if (dayTasks.length === 0) {

                taskContainer.innerHTML = `<div class="flex flex-col items-center justify-center py-16 text-gray-300"><div class="bg-gray-50 p-6 rounded-full mb-4"><i class="fas fa-feather-alt text-4xl"></i></div><p class="font-medium">No plans yet. Add a task!</p></div>`;

                return;

            }

            dayTasks.forEach(task => {

                const div = document.createElement('div');

                div.className = `flex items-center p-4 bg-white border border-gray-100 rounded-2xl shadow-sm transition-all active:scale-[0.98] ${task.completed ? 'bg-gray-50 opacity-60' : ''}`;

                div.innerHTML = `<div onclick="toggleTask(${task.id})" class="flex-1 flex items-center cursor-pointer"><div class="w-7 h-7 rounded-full border-2 mr-4 flex items-center justify-center transition-all ${task.completed ? 'bg-green-500 border-green-500' : 'border-gray-200'}">${task.completed ? '<i class="fas fa-check text-white text-xs"></i>' : ''}</div><span class="text-gray-700 text-sm ${task.completed ? 'line-through' : 'font-semibold'}">${task.text}</span></div><button onclick="deleteTask(${task.id})" class="text-gray-200 hover:text-red-400 p-2"><i class="fas fa-trash-alt text-sm"></i></button>`;

                taskContainer.appendChild(div);

            });

        }

        window.onload = () => {

            const today = new Date().toISOString().split('T')[0];

            if (today >= '2024-04-17' && today <= '2024-05-17') selectedDate = today;

            selectDate(selectedDate);

        };

    </script>
